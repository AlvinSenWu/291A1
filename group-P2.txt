CREATE TABLE Users (
    uid INTEGER,
    name CHAR(20),
    PRIMARY KEY (uid)
);

CREATE TABLE Artists (
    aid INTEGER,
    name CHAR(20),
    PRIMARY KEY (aid)
)

CREATE TABLE Songs (
    sid INTEGER,
    title CHAR(20),
    duration INTEGER,
    albid INTEGER,
    PRIMARY KEY (sid),
    FOREIGN KEY (aid) REFERENCES Artists
);

CREATE TABLE Albums (
    albid INTEGER,
    title CHAR(20),
    PRIMARY KEY (albid)
);

CREATE TABLE Al_Include (
    albid INTEGER,
    sid INTEGER,
    order INTEGER
    PRIMARY KEY (albid, sid),
    FOREIGN KEY (albid) REFERENCES Albums,
    FOREIGN KEY (sid) REFERENCES Songs
);

CREATE TABLE Playlists (
    plid INTEGER,
    title CHAR(20),
    uid INTEGER,
    PRIMARY KEY (plid),
    FOREIGN KEY (uid) REFERENCES Users
);

CREATE TABLE Pl_Include (
    plid INTEGER,
    sid INTEGER,
    order INTEGER
    PRIMARY KEY (plid, sid),
    FOREIGN KEY (plid) REFERENCES Playlists,
    FOREIGN KEY (sid) REFERENCES Songs
);

CREATE TABLE Sessions (
    sno INTEGER,
    sstart DATE,
    send DATE,
    sesOf uid,
    PRIMARY KEY (uid, sno),
    FOREIGN KEY (uid) REFERENCES Users
);

CREATE TABLE Played (
    sno INTEGER,
    uid INTEGER,
    sid INTEGER,
    cnt INTEGER,
    PRIMARY KEY (sno, uid, sid),
    FOREIGN KEY (sno) REFERENCES Songs,
    FOREIGN KEY (sno, uid) REFERENCES Sessions
);

CREATE TABLE Us_like (
    sid INTEGER,
    uid INTEGER,
    FOREIGN KEY (uid) REFERENCES Users,
    FOREIGN KEY (sid) REFERENCES Songs
);

CREATE TABLE Ua_like (
    albid INTEGER,
    uid INTEGER,
    FOREIGN KEY (uid) REFERENCES Users,
    FOREIGN KEY (albid) REFERENCES Albums
);

CREATE TABLE Up_like (
    plid INTEGER,
    uid INTEGER,
    FOREIGN KEY (uid) REFERENCES Users,
    FOREIGN KEY (plid) REFERENCES Pub_Playlists
);
